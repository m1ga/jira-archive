  3. Shuo Liang 2014-11-10 Hi, Just tested on iPhone 4 with IOS 7. Ti.Media.vibrate(); works well. Here is the test case. 
       var win = Titanium.UI.createWindow({  
       title:'Tab 1',
       backgroundColor:'#fff'
   });
   
   var button = Titanium.UI.createButton({
   	top: 50,
   	title: "vibrate start"
   });
   
   win.add(button);
    
   button.addEventListener("click", function(){
   	Ti.Media.vibrate();
   });
    
   win.open();
