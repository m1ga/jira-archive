Description

Given the following program 
var win = Titanium.UI.createWindow({  
    title:'Window',
    backgroundColor:'#fff',
    fullscreen: true
});


var tableData=[];

var row = Ti.UI.createTableViewRow({
	className: "grid",
        height: 120
});

var cell = Ti.UI.createView({
	
	width: 'auto',
	height: '120'

});
var label = Ti.UI.createLabel({
	width: 'auto',
	height:'auto',
	textAlign: 'center',
	text: 'Hello, World!',
	touchEnabled:false
}) 

cell.add(label);

row.add(cell);

tableData.push(row);

var tv = Ti.UI.createTableView({
data: tableData
});

tv.addEventListener('click',function(e){
	Ti.API.info(JSON.stringify(e.source));
});


tv.addEventListener('longclick',function(e){
	Ti.API.info(JSON.stringify(e.source));
	
});
win.add(tv);
win.open();
'click' event handler receives "[Ti.UI.View]" as source 'longclick' event handler receives "[Ti.UI.TableView]" as source
