  7. Paul Dowsett 2012-01-05 Sindre I realise that you are capable of writing this, but a colleague of mine kindly wrote it for you. It's not been tested, but I am sure it's almost there. Cheers 
       exports.log = function() {
       var log = '';
       for(var i = 0, len = arguments.length; i < len; i++){
           switch(typeof(arguments[i])) {
               case 'object':
               case 'array':
                   log += ' '+JSON.stringify(arguments[i]);
               break;
               case 'string':
               case 'number':
                   log += ' '+arguments[i];
               break;
               case 'boolean':
                   log += ' '+ (arguments[i]) ? 'true' : 'false';
               break;
               case 'undefined':
                   log += ' undefined';
               break;
               default:
                   log += ' null';
           }
       }
       Ti.API.info(log);
   };
