install! 'cocoapods',

        target.build_configurations.each do |config|
            config.build_settings['SWIFT_VERSION'] = '4.0'
        end
    end
  end

Both of these Pods are Swift pods which are supposed to be supported. Was there another regression with 2.2.2?
