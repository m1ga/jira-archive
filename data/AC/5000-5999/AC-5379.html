   picker.font= {
   
           fontSize: 10,
           
      };

   var column1 = Ti.UI.createPickerColumn({});
   
   Ti.API.info('Picket font is '+picker.font.fontSize);//undefined on Android
   
   for(var i=0, ilen=color.length; i
