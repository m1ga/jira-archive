code:

var win = Titanium.UI.createWindow({

title:"window",

});

var view = Ti.UI.createView({

backgroundColor:'red',
height:100,
width:100,
top:10

});

var a = Ti.UI.createAnimation();
a.center = {x:100,y:100}
a.duration = 1000;

view.addEventListener('click', function()
{

view.animate(a);

});

win.add(view);

win.open();
