Description

require('CrashWindow');
CrashWindow.js 
exports.createCrashWindow = function()
{
	var win = Ti.UI.createWindow();
	
	win.crash(arg){}; // invalid code
	
	return win;
};
Actual Result: Crash. Expected Result: 'Parse' Runtime Error.
