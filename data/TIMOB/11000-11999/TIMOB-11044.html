Description

Reproduction Steps: 1. Run the following code in an iOS 6 emulator: 
Ti.UI.setBackgroundColor('#000');
var win = Ti.UI.createWindow({
	backgroundColor : 'black',
	exitOnClose : true,
	fullscreen : false,
	title : 'TableView Demo'
});

var tableData = [];
var row = Ti.UI.createTableViewRow({
	className : 'forumEvent', // used to improve table performance
	selectedBackgroundColor : 'white',
	rowIndex : 1, // custom property, useful for determining the row during events
	height : 110
});

var labelUserName = Ti.UI.createLabel({
	color : '#576996',
	font : {
		fontFamily : 'Arial',
		fontSize : 16,
		fontWeight : 'bold'
	},
	text : 'Fred Smith ' + 1,
	left : 70,
	top : 6,
	width : 200,
	height : 30
});
row.add(labelUserName);
tableData.push(row);

var tableView = Ti.UI.createTableView({
	backgroundColor : 'white',
	data : tableData
});

win.addEventListener('click', function() {
	Ti.API.info('------------------------className: ' + row.className);
});

win.add(tableView);
win.open();
2. Click on the first row Expect Results: The log message should print out '------------------------className: forumEvent' Actual Results: The log message should print out '------------------------className: TiUITableViewRowProxy'
