alert('SESSION ID: ' + Ti.App.sessionId); 
Ti.API.error('SESSION ID: ' + Ti.App.sessionId); 
Ti.API.trace('1A. TRACE'); 
Ti.API.log('TRACE', '1B. TRACE'); 
Ti.API.debug('2A. DEBUG');

Ti.API.info('3A. INFO');
