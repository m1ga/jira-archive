Description

Description: Missing tableViewRows on Android. This is a regression, does not occur on 2.1.3 Steps: 1. Run the following code: 
var win = Ti.UI.createWindow({
    fullscreen : false,
    backgroundColor : 'white'
});
 
var tableView = Ti.UI.createTableView({
    top : '0dp',
    bottom : '0dp',
    left : '0dp',
    right : '0dp'
});
 
var numRows = 15;
var rows = [];
for (var i = 0; i < numRows; i++){
    var tableViewRow = Ti.UI.createTableViewRow({
        height : 'auto'
    });
     
    var textField = Ti.UI.createTextField({
        top:'0dp',
        left : '5dp',
        right : '5dp',
        borderColor : '#CCCCCC',
        borderStyle : Ti.UI.INPUT_BORDERSTYLE_BEZEL
    });
     
    tableViewRow.add(textField);
     
    rows.push(tableViewRow);
}
tableView.data = rows;
win.add(tableView);
win.open();
Expected: Should see a tableView with tableViewRows Actual: TableView is missing
