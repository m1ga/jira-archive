Description

Allow developers to query a list of tabs in a group by using the "tabs" property.

Acceptance Test

var tabGroup = Ti.UI.createTabGroup();
 
var tab1 = Ti.UI.createTab({
  title: 'A tab.',
  window: Ti.UI.createWindow({backgroundColor: 'blue'})
});
tabGroup.addTab(tab1);
 
var tab2 = Ti.UI.createTab({
  title: 'Another tab.',
  window: Ti.UI.createWindow({backgroundColor: 'red'})
});
tabGroup.addTab(tab2);

var tabs = tabGroup.tabs,
    count = tabs.length;
Ti.API.info('tabs count = ' + count);
Ti.API.info('tabs titles: ');
for (var i = 0; i < count; i++) {
	Ti.API.info(tabs[i].title);
}
 
tabGroup.open();
