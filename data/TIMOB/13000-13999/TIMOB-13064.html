Description

Problem description

When i call getValue() of date picker it returns null. It is necessary to store the date value form the picker 'change' event to be able to retrieve it.

Steps to reproduce

Use the following code to reproduce: 
var mainWindow = Ti.UI.createWindow({
    title : 'Lab',
    backgroundColor : 'white',
    fullscreen : false,
    navBarHidden : true,
    exitOnClose : true,
    orientationModes : [Titanium.UI.PORTRAIT]
});

var picker = Ti.UI.createPicker({
    type : Ti.UI.PICKER_TYPE_DATE,
    selectionIndicator : true,
    useSpinner : true
});

var showDate = Ti.UI.createButton({
    title : 'show date',
    bottom : 20
});
showDate.addEventListener('click', function(e) {
    alert(picker.getValue());
});

var view = Ti.UI.createView({
    width : Ti.UI.FILL,
    height : Ti.UI.FILL
});
view.add(picker);
view.add(showDate);

mainWindow.add(view);
mainWindow.open();
