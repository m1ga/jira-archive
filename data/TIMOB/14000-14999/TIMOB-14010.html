Description

*Problem* The image set on a button is not centered on Mobile Web. It's centered on iOS. *Test case* 
var win = Ti.UI.createWindow({
	title : 'test',
	backgroundColor : 'black',
});

var button = Ti.UI.createButton({
	backgroundColor : 'orange',
	image : 'KS_nav_ui.png',
	width : 300,
	height : 300
});

win.add(button);
win.open();
