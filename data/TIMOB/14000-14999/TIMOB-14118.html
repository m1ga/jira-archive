Description

Click event is supressed by touchstart event when invoked together. This is not regression since the issue occurs in 3.1.0 GA as well. Steps to reproduce: 1. Create an app using the code below. 2. click on the video. Expected: Consecutive alerts should alert 'click' and 'touchstart' events. Actual: Only 'touchstart' event is invoked. 

Titanium.UI.setBackgroundColor('#000');


var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});

var videoObject = Titanium.Media.createVideoPlayer({
		    contentURL:'movie.mp4',
		    scalingMode:Titanium.Media.VIDEO_SCALING_MODE_FILL,
		    top:0,
		    left:0,
		    height:'100%',
		    width:'100%',
		    autoplay:true
		});
		  
		videoObject.addEventListener('click', function(){
		   alert('click!');});
		   
		   videoObject.addEventListener('touchstart', function(){
		  alert('touchstart!');
		});
		
		win1.add(videoObject);
		win1.open();
