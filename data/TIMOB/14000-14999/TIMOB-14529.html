Description

*Problem* TextField focus doesn't work when TextField isn't visible on a screen. *Steps to reproduce* 1. Launch app 2. When you stay on the top of a TableView you can see first TextField there focus work without any problem 3. Try to hide shown keyboard for first TextField and scroll/move to the bottom of a TableView (There the focus doesn't work) *Test case* 
 
var win = Ti.UI.createWindow();
var table = Ti.UI.createTableView({
	style : Ti.UI.iPhone.TableViewStyle.GROUPED
});
var data = [];

for (var i = 0; i < 20; i++) {
	var row = Ti.UI.createTableViewRow({
		width : Ti.UI.FILL,
		height : Ti.UI.FILL
	});
	var tField = Ti.UI.createTextField({
		width : Ti.UI.FILL,
		height : "100%"
	});

	row.add(tField);
	data.push(row);
}

table.data = data;

win.add(table);
win.open();

setInterval(function() {
	data[0].children[0].focus();
}, 3000);
