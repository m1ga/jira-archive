Description

This code alerts "String empty" on Android and "String not empty" on iOS: 
var win = Ti.UI.createWindow({
	backgroundColor : 'white'
});

var lbl = Ti.UI.createLabel();

if (lbl.text === "") {
	alert('String empty');
} else {
	alert('String not empty');
}

win.add(lbl);

win.open();
