Description

The animation system crashes when there are two animations in play for one element and the first animation stops before the second one starts because the second one had a delay set. 
/*global Ti*/

var win = Ti.UI.createWindow({
		backgroundColor: '#fff'
	}),
	view = Ti.UI.createView({
		backgroundColor: '#f00',
		width: 100,
		height: 100,
		left: 200,
		top: 200
	});

win.add(view);
win.open();

require.on.once(view, 'postlayout', function () {
	setTimeout(function () {
		view.animate({
			left: 300,
			width: 100,
			height: 100,
			duration: 500
		});

		setTimeout(function () {
			view.animate({
				top: 300,
				width: 100,
				height: 100,
				duration: 500,
				delay: 200
			});
		}, 400);
	}, 0);
});
