Description

  2. Vishal Duggal 2014-10-01 Sample test case. Change the values of the creation dictionary to see behavior and UI changes. 
       var win = Ti.UI.createWindow({backgroundColor:'white'});
   
   var btn = Ti.UI.createButton({title:'ALERT'});
   
   win.add(btn);
   
   win.open();
   
   
   function clickhandler(e){
   	Ti.API.info(JSON.stringify(e));
   }
   
   
   btn.addEventListener('click',function(e){
   	var alertDialog = Ti.UI.createAlertDialog({
   		title:'Alert!',
   		message:'Hello World!',
   		buttonNames:['OK','Cancel'],
   		cancel:1,
   		style:3,
   		persistent:true
   	});
   	alertDialog.addEventListener('click',clickhandler)
   	alertDialog.show();
   })
