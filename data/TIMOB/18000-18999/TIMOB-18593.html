Description

The following example shows that setting a MenuItem title via applyProperties doesn't work, while setting it on creation or via the property or its setter does work. The applyProperties is listed in documentation as being available for Android for this proxy. 
var win = Ti.UI.createWindow({
  fullscreen: true
});

win.activity.onCreateOptionsMenu = function(e){
  var menu = e.menu;
  var menuItem = menu.add({ 
    title: "On create", 
    showAsAction: Ti.Android.SHOW_AS_ACTION_ALWAYS
  });
  menuItem.applyProperties({
    title: 'via applyProperties'
  });
  // menuItem.title = 'via property';
  // menuItem.setTitle('via setter');
};

win.open();
