  6. Hans Knöchel 2016-01-05 PR: https://github.com/appcelerator-modules/ti.barcode/pull/32 Also fixes MOD-2183. Example: 
       var Barcode = require('ti.barcode');
   
   var win = Ti.UI.createWindow({
       title: 'Test',
       backgroundColor: '#eee',
       statusBarStyle: Titanium.UI.iPhone.StatusBar.LIGHT_CONTENT
   });
   
   var overlay = Ti.UI.createView({
       backgroundColor: '#f00',
       top: 0, left: 0, right: 0,
       height: 50
   });
   
   var header = Ti.UI.createView({
       backgroundColor: '#222',
       top: 0,
       height: 50
   });
   overlay.add(header);
   
   //win.add(overlay);
   win.addEventListener('open', function () {
       Barcode.capture({
           animate: false,
           overlay: overlay,
           showCancel: true,
           showRectangle: true,
           keepOpen: true,
           acceptedFormats: [
               Barcode.FORMAT_QR_CODE
           ]
       });
   });
   
   Barcode.addEventListener('success', function (e) {
       Ti.API.info('Barcode: ' + e.result);
   });
   
   win.open();
