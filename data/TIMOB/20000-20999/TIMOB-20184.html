var win = Ti.UI.createWindow({backgroundColor : "white"});

    width: Ti.UI.SIZE,
    title: "I'm loooooooooooooong button",
    top : 0,
    height : 50
});

    width: Ti.UI.SIZE,
    title: "s",
    top : 60,
    height : 50
});

    width: Ti.UI.SIZE,
    title: "short",
    top : 120,
    height : 50

win.add(button3);

win.open();
