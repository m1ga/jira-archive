       var textField = Ti.UI.createTextField({
     borderStyle: Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
     color: '#336699',
     borderColor : '#000',
     backgroundColor : 'transparent',
     top: 100, left: 10,
     width: 250,
     font: { fontSize: '14dp' },
   });
