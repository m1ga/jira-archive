Description

	backgroundColor: "white"
}
"#imageContainer" : {
	height: Ti.UI.SIZE, width: Ti.UI.FILL
}
"#large" : {
	image: 'images/large.png',
	height: Ti.UI.SIZE, width: Ti.UI.FILL
}
"#large[platform=android]" : {
	image: '../../images/large.png'
}
"#small" : {
	top: 0, right: 0,
	image: 'images/small.png',
	height: Ti.UI.SIZE, width: 100
}
"#small[platform=android]" : {
	image: '../../images/small.png'
}
