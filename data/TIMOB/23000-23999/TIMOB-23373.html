Description

Label with height should set label height accordingly and render text vertically center by default. 
var win = Ti.UI.createWindow({ backgroundColor: 'white' });

var view = Ti.UI.createLabel({
    text: "ABC",
    height: 100,
    width: 100,
    color: "#C41230",
    backgroundColor: "#F2D0D6",
    textAlign: "center",
});

win.add(view);
win.open();
