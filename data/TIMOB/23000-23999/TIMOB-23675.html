Description

Comments

  1. Hans Knöchel 2016-07-24 PR: https://github.com/appcelerator/titanium_mobile/pull/8153 Demo: 
       var win = Ti.UI.createWindow({
       backgroundColor: "#fff"
   });
   var btn = Ti.UI.createButton({
       title: "Show alert"
   });
   
   btn.addEventListener("click", function() {
       var alert = Ti.UI.createAlertDialog({
           title: "Test",
           message: "Message",
           buttonNames: ["OK", "Cancel", "Destructive"],
           cancel: 1,
           destructive: 2,
           preferred: 0
       });
       alert.addEventListener("click", function(e) {
           Ti.API.warn(e);
       })
       alert.show();
   });
   
   win.add(btn);
   win.open();
