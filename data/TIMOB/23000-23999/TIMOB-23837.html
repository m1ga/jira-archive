  1. Hans Knöchel 2016-08-29 PR: https://github.com/appcelerator/titanium_mobile/pull/8286 Use the attached "chat@2x.png" for the icon that is required for badges - see TIMOB-8880 for possible issue with that. Demo: 
       var tabGroup = Ti.UI.createTabGroup({
       unselectedItemTintColor: "red",
       tabs: [createTab("Tab 1"), createTab("Tab 2"), createTab("Tab 3"), createTab("Tab 4")]
   });
   
   // Sets the badge after 3 seconds
   setTimeout(function() {
       tabGroup.tabs[1].badge = 3;
       tabGroup.tabs[1].badgeColor = "gray";
   },2000);
   
   tabGroup.open();
   
   function createTab(title) {
       return Ti.UI.createTab({
           title: title,
           icon: "chat.png",
           window: Ti.UI.createWindow({
               title: title
           })
       })
   }
