Description

When parent View has border, child Views calculates its position and size based on parent border. 
var win = Ti.UI.createWindow(),
    view0 = Ti.UI.createView({
        top: 20,
        bottom: 20,
        left: 20,
        right: 20,
        borderColor: 'blue',
        borderWidth: 1,
        backgroundColor: 'white'
    }),
    view = Ti.UI.createView({
        top: 10,
        bottom: 10,
        left: 10,
        right: 10,
        borderColor: 'green',
        borderWidth: 10,
        backgroundColor: 'red'
    });
view.add(view0);

win.add(view);
win.open();
