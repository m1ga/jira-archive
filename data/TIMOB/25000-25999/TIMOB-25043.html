Description

The method isValid() in Titanium.Network.Cookie is missing on Windows. Test Case 
var cookie1 = Ti.Network.createCookie({
	domain: 'example.com',
	name: 'test_cookie',
	value: '12',
	path: '/'
});

var cookie2 = Ti.Network.createCookie({
	name: 'test_cookie',
	value: '12',
	path: '/'
});

console.log('is cookie valid: ' + cookie1.isValid());
console.log('is cookie valid: ' + cookie2.isValid());
