    backgroundColor: '#fff'
});

var btnTakeMovie = Ti.UI.createButton({ title: 'title_take_video', top: 10, left: 5 });

btnTakeMovie.addEventListener('click', function (e) {
    Titanium.Media.showCamera({

            Ti.API.info('success');
