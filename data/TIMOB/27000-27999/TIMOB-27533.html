Hello! TabbedBar backgroundColor is not working with the latest SDK on iOS. *Test Code*: 
var win = Ti.UI.createWindow({
    layout : "vertical"
});

var bb1 = Titanium.UI.createTabbedBar({
    labels : ['One', 'Two', 'Three'],
    backgroundColor : 'red',
    top : 50,
    height : 25,
    width : 200
});
win.add(bb1);
win.open();
