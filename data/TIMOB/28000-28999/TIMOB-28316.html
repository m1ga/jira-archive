var win = Ti.UI.createWindow({ backgroundColor: "white", fullscreen: true });
  var button = Ti.UI.createButton({
    borderRadius: "22 22",
    height: 44,
    width: 180,
    backgroundColor: "#eceded",
    title: "Click me",
    viewShadowColor: "4d000000",
    viewShadowOffset: {
      y: 2,
      x: 2,
    },
    viewShadowRadius: 5,
  });

  button.addEventListener("click", () => {
    button.hide();
    setTimeout(() => {
      button.show();
    }, 2000);
  });
  win.add(button);
  win.open();
