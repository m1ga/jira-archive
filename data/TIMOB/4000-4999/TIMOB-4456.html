Description

Add the ability to do the following: 
var win = Ti.UI.createWindow({backgroundColor:'white'});
var searchBar = Ti.UI.createSearchBar();
var tableview = Ti.UI.createTableView({
	search: searchBar
});
win.add(tableview);
setTimeout( function() {
	searchBar.focus();
	setTimeout( function() {
	        searchBar.value = 'dog';
	}, 500);
}, 500);
setTimeout( function() {
	alert('fired!');
        
        // This is the method desired
	searchBar.cancel();
}, 3000);
win.open();
