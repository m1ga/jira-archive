  1. Bacto 2011-09-12 I'd like too ! It will be very good if you add this :-)
  2. Pedro Enrique 2013-11-19 This will be possible in 3.2.0 using the Ti.UI.iOS.AttributedString() component You can download for the CI build and try it out. 
       var win = Ti.UI.createWindow({
       backgroundColor: '#CCC'
   });
   
   var hintText = 'This is a hint text';
   var attributedString = Ti.UI.iOS.createAttributedString({
       attributes: [
           {
               type: Ti.UI.iOS.ATTRIBUTE_FOREGROUND_COLOR,
               value: 'blue', 
               range: [0, hintText.length]
   
           }
       ],
       text: hintText
   });
   
   var textField = Ti.UI.createTextField({
       attributedHintText: attributedString,
       left: 10,
       right: 10,
       borderStyle: Ti.UI.INPUT_BORDERSTYLE_ROUNDED
   });
   
   win.add(textField);
    
   win.open();
