var win = Titanium.UI.createWindow({});
var iv = Ti.UI.createImageView({
  image: 'checked.png',
  top: 100
});
var webview = Titanium.UI.createWebView({
    html: '<html><body><img src="checked.png"></body></html>',  
});
win.add(webview);
win.add(iv);
win.open();
