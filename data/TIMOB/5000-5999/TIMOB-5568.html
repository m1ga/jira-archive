	title: 'title ch1'
});

	title: 'title ch2'
});

win.add(b1);
win.add(b2);

b1.addEventListener('click', function(){
	win.title = 'It is the TITLE';
});

b2.addEventListener('click', function(){
	win.title = 'It is the SUPERTITLE';
});
