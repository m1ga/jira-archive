Description

Can't represent event 'error' in webView. Event 'load' fired when loading any type of URL. 
var win = Ti.UI.currentWindow;

var tf = Ti.UI.createTextField({
	left: 10,
	top: 5,
	width: 320,
	height: 40
});

var webView = Ti.UI.createWebView({
	top: 50,
	left: 10,
	width: 300,
	height: 300,
	url: 'google.com/404',
	backgroundColor: 'red'
});

win.add(tf);
win.add(webView);

tf.addEventListener('return',function(){
	webView.url = tf.value;
});

webView.addEventListener('beforeload', function(){
	alert('Page loading begins');
});

webView.addEventListener('load',function(){
	alert('Page loaded');
});

webView.addEventListener('error',function(){
	alert('page cannot be load');
});
