Description

The click event not works with slider object in desktop browser. The dblclick event works good. 
var win = Ti.UI.currentWindow;

var slider = Ti.UI.createSlider({
	value: 'the slider',
	top: 10,
	left: 10,
	width: 200,
	height: 50,
	min: 0,
	max: 100
});


win.add(slider);

slider.addEventListener('click', function(){
	alert('click works');
});

slider.addEventListener('dblclick', function(){
	alert('dblclick works');
});
