Description

If you try to open the emailDialog with animated true it crashes the app. Step 1: run the code below Step 2: click on the screen Step 3: see the app crash Step 4: remove '{animated:true}' from the open Step 5: run again Step 6: see the dialog open as it should 
var win = Ti.UI.createWindow({
	backgroundColor:'blue'
});

win.addEventListener('click',function(e){
	// works fine with modal windows
	// var newWin = Ti.UI.createWindow({modal:true});
	// newWin.open({animated:false});
	
	var email = Ti.UI.createEmailDialog();
	email.open({animated:true});
});

win.open();
