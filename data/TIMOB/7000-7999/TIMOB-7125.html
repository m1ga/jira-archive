Description

Problem

Incorrect left padding for first child using horizontal layout. See attached screenshot.

Test case

var win = Ti.UI.createWindow({
  title:'Window 1',
  backgroundColor:'#fff'
});
var view = Ti.UI.createView({
  layout: 'horizontal'
});

for(var x=0;x<10;x++){
  var subview = Ti.UI.createButton({
    top: 10,
    left: 10,
    width: 100,
    height: 100,
    title: 'Test'
  });
  view.add(subview);
}

win.add(view);
win.open();
