	textAlign:'center',

});

win1.add(label1);

    backgroundColor:'#fff'
});
var tab2 = Titanium.UI.createTab({  
    icon:'KS_nav_ui.png',
    title:'Tab 2',
    window:win2
});

var label2 = Titanium.UI.createLabel({
	color:'#999',
	text:'I am Window 2',
	font:{fontSize:20,fontFamily:'Helvetica Neue'},
	textAlign:'center',
	width:'auto'
});

win2.add(label2);


var win3 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});

    icon:'KS_nav_views.png',
    title:'Tab 1',
    window:win3
});

var label3 = Titanium.UI.createLabel({
