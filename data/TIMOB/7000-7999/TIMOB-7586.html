Description

Problem

Saving an image to the filesystem using toImage() method, does not keep the transparent property of the image

Steps to reproduce

See attached code 
var window = Ti.UI.createWindow({
    backgroundColor:'#ff0'
});

var view = Ti.UI.createImageView({
    image:'KS_nav_ui.png',
// var view = Ti.UI.createView({
    backgroundColor:'transparent',
    borderWidth:2,
    borderColor:'#f00',
    width:100,
    height:100,
    top:10
});

window.add(view);

var image = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory, 'tst.png');
image.write(view.toImage());

var img = Ti.UI.createImageView({
    width:200,
    height:200,
    bottom:10,
    image:image.nativePath
});

window.add(img);
window.open();
