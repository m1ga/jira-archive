{
	"deploytype": "development",
	"guid": "667ea9ab-0f6a-46e7-b66d-346662f4ec97",
	"appid": "com.test.test1",
	"platform": "mobileweb",
	"modules": [
		{
			"id": "com.test.module1",
			"guid": "23d30f94-feb6-4a04-b440-dc111e9e6e8b",
			"author": "Module Developer ",
			"description": "Test Module 1",
			"name": "test1",

			"license": "Apache",
			"version": "1.0"
		}
	],
	"version": "1.0",
	"name": "Test 1",
	"success": true,
	"code": 0,
	"reason": "good to go"
}
