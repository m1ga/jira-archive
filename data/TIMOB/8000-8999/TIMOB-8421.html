Description

When adding a view with a size specified in points (mm), the view is sized in centimeters, instead. In the following code sample, the blue and red boxes should be the same size, as on iOS. See attached screenshots. 
var win1 = Titanium.UI.createWindow({  
    title:'Composite Layout Test',
    backgroundColor:'#fff',
    layout: 'vertical'
});
var l1 = Titanium.UI.createLabel({
	top: 0,
	color:'#999',
	backgroundColor: 'blue',
	text:'1in',
	font:{fontSize:20,fontFamily:'Helvetica Neue'},
	textAlign:'center',
	width:'1in',
	height:'1in'
});
win1.add(l1);

var l2 = Titanium.UI.createLabel({
	top: 0,
	color:'#999',
	backgroundColor: 'red',
	text:'72pt',
	font:{fontSize:20,fontFamily:'Helvetica Neue'},
	textAlign:'center',
        // this should be 1 inch
	width:'25.4mm',
	height:'25.4mm'
});
win1.add(l2);

var l3 = Titanium.UI.createLabel({
	top: 0,
	color:'#999',
	backgroundColor: 'green',
	text:'72 units',
	font:{fontSize:20,fontFamily:'Helvetica Neue'},
	textAlign:'center',
	width:72,
	height:72
});
win1.add(l3);

win1.open();
