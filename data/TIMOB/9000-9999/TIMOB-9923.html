Description

Discovered while doing some parity testing. Drop the following code into app.js and click the SIZE buttons for width or height. Observed: Label disappears. Expected: Label does not disappear 

var win = Ti.UI.createWindow({
    backgroundColor: 'white',
});

var theFont = {fontSize: 11, font: 'Helevetica' };
var container = Ti.UI.createView({
	layout:'vertical'
}) 
var controls1 = Ti.UI.createView({
	layout:'horizontal',
	height:Ti.UI.SIZE,
	top:10,
	borderColor:'#ccc',
	borderWidth:1
})

	font :theFont,
	text:'V-ALIGN'
